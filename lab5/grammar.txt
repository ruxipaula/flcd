PROGRAM ~ start { STMTLIST }
DECLARATION ~ TYPE IDENTIFIER
TYPE1 ~ int | string
ARRAYDECL ~ TYPE1 [ INTEGER ]
TYPE ~ TYPE1 | ARRAYDECL
STMTLIST ~ STMT | STMT ; STMTLIST
STMT ~ SIMPLSTMT | STRUCTSTMT | DECLARATION
SIMPLSTMT ~ ASSIGNSTMT | IOSTMT
ASSIGNSTMT ~ IDENTIFIER = EXPRESSION
EXPRESSION ~ EXPRESSION + TERM | EXPRESSION - TERM | TERM
TERM ~ TERM * FACTOR | TERM / FACTOR | FACTOR
FACTOR ~ ( EXPRESSION ) | IDENTIFIER | CONSTCHAR
IOSTMT ~ readint ( ) | readstring ( ) | print ( IDENTIFIER )
STRUCTSTMT ~ CMPDSTMT | IFSTMT | WHILESTMT
WHILESTMT ~ while ( CONDITION ) { STMT }
CMPDSTMT ~ { STMTLIST }
IFSTMT ~ if ( CONDITION ) STMT | if ( CONDITION ) STMT else STMT
FORSTMT ~ for ( ASSIGNSTMT ; CONDITION ; ASSIGNSTMT ) STMT
CONDITION ~ EXPRESSION RELATION EXPRESSION
RELATION ~ < | <= | == | != | >= | > | &&
IDENTIFIER ~ LETTER SEQUENCE_OF_LETTERS_OR_DIGITS
SEQUENCE_OF_LETTERS_OR_DIGITS ~ DIGIT_OR_LETTER | DIGIT_OR_LETTER SEQUENCE_OF_LETTERS_OR_DIGITS
